Expand module

With this module you can expand any item. It requires a container div, and within this container you can add 2 triggers: `trigger` & `target`. See implementation of this code below:

Example

How to use

Html

    <div class="expand expand--container">
        <h3 class="expand--trigger">This is the title</h3>
        <span class="expand--target">Lorem ipsum dolor sit amet, consectetur adipisicing elit. </span>
    </div>

Css

    /* UX for clicking on trigger */
    .expand--trigger {
    	cursor: pointer;
    }

    /* Show target when expand element is open */
    .expand__isOpen .expand--target {
        opacity: 1;
    }

    /* Hide target when expand element is closed */
    .expand__isClosed .expand--target {
        opacity: 0;
    }

    /* Custom Styling */
    .expand--target {
    	transition: .4s margin linear,.2s opacity linear;
    	float: left;
    	display: inline-block;
    }

    .expand__isClosed .expand--target {
        margin-top: -20%;
    }
Javascript


    Expand.prototype.init = function(){
        var expand = this;
        expand.status = true;

        this.configureTrigger("trigger", function(trigger) {

            trigger.addAction('click',function(){
                if (expand.status) {
                    expand.setState("Closed");
                    expand.removeState("Open");
                    expand.status = false;
                } else {
                    expand.setState("Open");
                    expand.removeState("Closed");
                    expand.status = true;
                }
            }, {
                addDefaults:false
            })

        });
    }
